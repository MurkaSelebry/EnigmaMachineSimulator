190 private byte[] ObjectToByteArray(Object obj)
191 {
192 if (obj == null)
193 return null;
194 BinaryFormatter bf = new BinaryFormatter();
195 MemoryStream ms = new MemoryStream();
196 bf.Serialize(ms, obj);
197 return ms.ToArray();
198 }
199 private Object ByteArrayToObject(byte[] arrBytes)
200 {
201 MemoryStream memStream = new MemoryStream();
202 BinaryFormatter binForm = new BinaryFormatter();
203 memStream.Write(arrBytes, 0, arrBytes.Length);
204 memStream.Seek(0, SeekOrigin.Begin);
205 Object obj = (Object)binForm.Deserialize(memStream);
206 return obj;
207 }
